667 mat4 out_matrix = mat4_identity();
668
669 const f32 *m1_ptr = matrix_0.data;

703OINLINE mat4 mat4_orthographic(f32 left, f32 right, f32 bottom, f32 top,
704 f32 near, f32 far) {
705 mat4 out_matrix = mat4_identity();
706
707 f32 lr = 1.0f / (left - right);
708 f32 bt = 1.0f / (bottom - top);
709 f32 nf = 1.0f / (near - far);
710
711 out_matrix.data[0] = -2.0f * lr;
712 out_matrix.data[5] = -2.0f * bt;
713 out_matrix.data[10] = 2.0f * nf;
714
715 out_matrix.data[12] = (left + right) * lr;
716 out_matrix.data[13] = (top + bottom) * bt;
717 out_matrix.data[14] = (far + near) * nf;
718 return out_matrix;
719}

731OINLINE mat4 mat4_perspective(f32 fov_radians, f32 aspect_ratio, f32 near,
732 f32 far) {

734 mat4 out_matrix;

736 out_matrix.data[0] = 1.0f / (aspect_ratio * half_tan_fov);

753OINLINE mat4 mat4_look_at(vec3 position, vec3 target, vec3 up) {
754 mat4 out_matrix;
755 vec3 z_axis;
756 z_axis.x = target.x - position.x;
757 z_axis.y = target.y - position.y;
758 z_axis.z = target.z - position.z;
759
760 z_axis = vec3_normalized(z_axis);
761 vec3 x_axis = vec3_normalized(vec3_cross(z_axis, up));
762 vec3 y_axis = vec3_cross(x_axis, z_axis);
763
764 out_matrix.data[0] = x_axis.x;
765 out_matrix.data[1] = y_axis.x;
766 out_matrix.data[2] = -z_axis.x;
767 out_matrix.data[3] = 0;
768 out_matrix.data[4] = x_axis.y;
769 out_matrix.data[5] = y_axis.y;
770 out_matrix.data[6] = -z_axis.y;
771 out_matrix.data[7] = 0;
772 out_matrix.data[8] = x_axis.z;
773 out_matrix.data[9] = y_axis.z;
774 out_matrix.data[10] = -z_axis.z;
775 out_matrix.data[11] = 0;
776 out_matrix.data[12] = -vec3_dot(x_axis, position);
777 out_matrix.data[13] = -vec3_dot(y_axis, position);
778 out_matrix.data[14] = vec3_dot(z_axis, position);
779 out_matrix.data[15] = 1.0f;
780
781 return out_matrix;
782}

1094OINLINE mat4 quat_to_mat4(quat q) {
1095 mat4 out_matrix = mat4_identity();
1096
1097 // https://stackoverflow.com/questions/1556260/convert-quaternion-rotation-to-rotation-matrix
1098
1099 quat n = quat_normalize(q);
1100
1101 out_matrix.data[0] = 1.0f - 2.0f * n.y * n.y - 2.0f * n.z * n.z;
1102 out_matrix.data[1] = 2.0f * n.x * n.y - 2.0f * n.z * n.w;
1103 out_matrix.data[2] = 2.0f * n.x * n.z + 2.0f * n.y * n.w;
1104
1105 out_matrix.data[4] = 2.0f * n.x * n.y + 2.0f * n.z * n.w;
1106 out_matrix.data[5] = 1.0f - 2.0f * n.x * n.x - 2.0f * n.z * n.z;
1107 out_matrix.data[6] = 2.0f * n.y * n.z - 2.0f * n.x * n.w;
1108
1109 out_matrix.data[8] = 2.0f * n.x * n.z - 2.0f * n.y * n.w;
1110 out_matrix.data[9] = 2.0f * n.y * n.z + 2.0f * n.x * n.w;
1111 out_matrix.data[10] = 1.0f - 2.0f * n.x * n.x - 2.0f * n.y * n.y;
1112
1113 return out_matrix;
1114}

1116// Calculates a rotation matrix based on the quaternion and the passed in center
1117// point.
1118OINLINE mat4 quat_to_rotation_matrix(quat q, vec3 center) {
1119 mat4 out_matrix;
1120
1121 f32 *o = out_matrix.data;
1122 o[0] = (q.x * q.x) - (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1123 o[1] = 2.0f * ((q.x * q.y) + (q.z * q.w));
1124 o[2] = 2.0f * ((q.x * q.z) - (q.y * q.w));
1125 o[3] = center.x - center.x * o[0] - center.y * o[1] - center.z * o[2];
1126
1127 o[4] = 2.0f * ((q.x * q.y) - (q.z * q.w));
1128 o[5] = -(q.x * q.x) + (q.y * q.y) - (q.z * q.z) + (q.w * q.w);
1129 o[6] = 2.0f * ((q.y * q.z) + (q.x * q.w));
1130 o[7] = center.y - center.x * o[4] - center.y * o[5] - center.z * o[6];
1131
1132 o[8] = 2.0f * ((q.x * q.z) + (q.y * q.w));
1133 o[9] = 2.0f * ((q.y * q.z) - (q.x * q.w));
1134 o[10] = -(q.x * q.x) - (q.y * q.y) + (q.z * q.z) + (q.w * q.w);
1135 o[11] = center.z - center.x * o[8] - center.y * o[9] - center.z * o[10];
1136
1137 o[12] = 0.0f;
1138 o[13] = 0.0f;
1139 o[14] = 0.0f;
1140 o[15] = 1.0f;
1141 return out_matrix;
1142}

1156// slerp = Spherical Linear Interpolation
1157OINLINE quat quat_slerp(quat q_0, quat q_1, f32 percentage) {
1158 quat out_quaternion;
1159 // Source: https://en.wikipedia.org/wiki/Slerp
1160 // Only unit quaternions are valid rotations.
1161 // Normalize to avoid undefined behavior.
1162 quat v0 = quat_normalize(q_0);
1163 quat v1 = quat_normalize(q_1);
1164
1165 // Compute the cosine of the angle between the two vectors.
1166 f32 dot = quat_dot(v0, v1);
1167
1168 // If the dot product is negative, slerp won't take
1169 // the shorter path. Note that v1 and -v1 are equivalent when
1170 // the negation is applied to all four components. Fix by
1171 // reversing one quaternion.
1172 if (dot < 0.0f) {
1173 v1.x = -v1.x;
1174 v1.y = -v1.y;
1175 v1.z = -v1.z;
1176 v1.w = -v1.w;
1177 dot = -dot;
1178 }
1179
1180 const f32 DOT_THRESHOLD = 0.9995f;
1181 if (dot > DOT_THRESHOLD) {
1182 // If the inputs are too close for comfort, linearly interpolate
1183 // and normalize the result.
1184 out_quaternion = (quat){v0.x + ((v1.x - v0.x) * percentage),
1185 v0.y + ((v1.y - v0.y) * percentage),
1186 v0.z + ((v1.z - v0.z) * percentage),
1187 v0.w + ((v1.w - v0.w) * percentage)};
1188
1189 return quat_normalize(out_quaternion);
1190 }
1191
1192 // Since dot is in range [0, DOT_THRESHOLD], acos is safe
1193 f32 theta_0 = oacos(dot); // theta_0 = angle between input vectors
1194 f32 theta = theta_0 * percentage; // theta = angle between v0 and result
1195 f32 sin_theta = osin(theta); // compute this value only once
1196 f32 sin_theta_0 = osin(theta_0); // compute this value only once
1197
1198 f32 s0 =
1199 ocos(theta) -
1200 dot * sin_theta / sin_theta_0; // == sin(theta_0 - theta) / sin(theta_0)
1201 f32 s1 = sin_theta / sin_theta_0;
1202
1203 return (quat){(v0.x * s0) + (v1.x * s1), (v0.y * s0) + (v1.y * s1),
1204 (v0.z * s0) + (v1.z * s1), (v0.w * s0) + (v1.w * s1)};
1205}
